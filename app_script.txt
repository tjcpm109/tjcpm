/**
 * TJCPM 員工系統 - Google Apps Script 後端
 *
 * 修正紀錄：
 * 1. doGet() 原本沒有 'approve' 分支：前端 approveRecord() 是用 callGAS()（JSONP，
 *    本質上是 GET），但審核邏輯只寫在 doPost() 裡，所以主管按下核准/拒絕後，
 *    請求根本沒有打到審核程式碼，狀態當然就不會寫回 Sheet。
 *    → 修正：把審核邏輯抽成 handleApprove()，doGet/doPost 都呼叫同一份。
 * 2. 原比對邏輯用 empId + date 字串比對，當同一人同一天有多筆同類型申請時
 *    會比對到錯誤的列、或日期格式（'2026/6/25' vs '2026-06-25'）不一致直接比對失敗。
 *    → 修正：優先用 clientId 精準比對，找不到才 fallback 用 empId + 狀態=待審 模糊比對。
 * 3. 新增「補打卡」工作表的讀寫，對應前端新增的補打卡頁面。
 * 4. 新增：請假／加班／補打卡申請送出後，自動寄信通知主管查閱審核。
 * 5. ★ 修正欄位定位 bug：原本用「動態抓最後一欄」來推算狀態欄位置
 *    （statusCol = lastCol - 1），但這個算法實際上算到 clientId 欄，
 *    導致核准/拒絕的結果被寫進 clientId 欄、狀態欄反而完全沒被更新。
 *    → 修正：改用下面 SHEET_COLUMNS 的固定欄位編號，不再依賴動態推算。
 * 6. 新增：審核時一併記錄「核准人姓名」到 Sheet。
 * 7. 新增：appendToSheetSafely() — 找不到對應分頁、或寫入失敗時不會讓整個請求中斷，
 *    只記錄到 Apps Script 執行紀錄；並把「寫入 Sheet」跟「寄信通知」拆開，
 *    即使 Sheet 寫入失敗，主管還是會收到通知信，方便判斷問題出在哪一段。
 * 8. 新增：action=diag 自助診斷端點，直接用瀏覽器打開 GAS_URL?action=diag
 *    即可看到目前 Sheet 裡實際有哪些分頁、欄位標題，不用回頭看程式碼比對。
 */

// 請假／加班／補打卡 有新申請時，統一通知這些主管信箱
const ADMIN_NOTIFY_EMAILS = ['acls2000@tmu.edu.tw', 'juchi.tmu@gmail.com'];

// 各申請工作表的固定欄位編號（1-indexed，對應 sheet.getRange 的欄號）。
// 重要：appendRow() 寫入的順序必須跟這裡的編號完全對應，新增欄位時兩邊要一起改。
//
// 請假申請：A時間 B員編 C姓名 D假別 E日期(起) F日期(迄) G時間(起) H時間(迄) I（空） J審核狀態 K clientId L核准人姓名 M簽核時間 N是否同意
// 加班申請：A時間 B員編 C姓名 D開始日期 E結束日期 F開始時間 G結束時間 H原因 I狀態 J clientId K核准人姓名
// 補打卡申請：A時間 B員編 C姓名 D類型 E日期 F時間 G原因 H狀態 I clientId J核准人姓名
const SHEET_COLUMNS = {
  '請假申請':   { status: 10, clientId: 11, approver: 12, approveTime: 13 },
  '加班申請':   { status: 9,  clientId: 10, approver: 11, approveTime: 12 },
  '補打卡申請': { status: 8,  clientId: 9,  approver: 10, approveTime: 11 }
};

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const now = new Date();

    // 審批（保留 POST 入口，相容舊版呼叫方式）
    if (data.action === 'approve') {
      const result = handleApprove(ss, data);
      return respond(result);
    }

    // 登入驗證
    if (data.action === 'login') {
      return respond(handleLogin(ss, data));
    }

    // 打卡
    if (data.type === '上班' || data.type === '下班') {
      appendToSheetSafely(ss, '打卡紀錄', [now, data.empId, data.name || '', data.type, data.date, data.time, data.lat || '', data.lng || '', data.clientId || '']);
    }

    // 請假（欄位順序對應 SHEET_COLUMNS['請假申請']：最後留一欄空白給核准人姓名）
    // ★ 寫入 Sheet 跟寄信通知分開處理：就算工作表找不到、寫入失敗，也還是會嘗試寄信，
    //   這樣才能分辨「申請真的有送到伺服器但存不進 Sheet」跟「申請根本沒送到伺服器」這兩種情況。
    else if (data.type === '請假') {
      // 欄位順序對應 Sheet：A時間 B員編 C姓名 D假別 E日期(起) F日期(迄) G時間(起) H時間(迄) I（空白欄） J審核狀態 K clientId L核准人（空）
      appendToSheetSafely(ss, '請假申請', [now, data.empId, data.name || '', data.subType || '', data.date, data.endDate || '', data.startTime || '', data.endTime || '', '', '待審', data.clientId || '', '']);
      notifyAdmins('請假', data);
    }

    // 加班
    else if (data.type === '加班') {
      appendToSheetSafely(ss, '加班申請', [now, data.empId, data.name || '', data.date, data.startTime || '', data.endTime || '', data.hours || '', data.reason || '', '待審', data.clientId || '', '']);
      notifyAdmins('加班', data);
    }

    // 補打卡
    else if (data.type === '補打卡') {
      appendToSheetSafely(ss, '補打卡申請', [now, data.empId, data.name || '', data.subType || '', data.date, data.time || '', data.reason || '', '待審', data.clientId || '', '']);
      notifyAdmins('補打卡', data);
    }

    return respond({ status: 'ok' });

  } catch (err) {
    return respond({ status: 'error', message: err.toString() });
  }
}

function doGet(e) {
  const callback = e.parameter.callback;
  const params = e.parameter.data ? JSON.parse(e.parameter.data) : {};
  const action = params.action || e.parameter.action;
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  let result = { status: 'ok' };

  try {
    // 登入驗證（JSONP）
    if (action === 'login') {
      result = handleLogin(ss, params);
    }

    // ★ 審批（JSONP）— 這是前端真正會打到的入口，之前少了這個分支，
    //    導致主管核准/拒絕永遠不會寫回 Sheet。
    else if (action === 'approve') {
      result = handleApprove(ss, params);
    }

    // 取得所有打卡紀錄
    else if (action === 'getAll') {
      const sheet = ss.getSheetByName('打卡紀錄');
      result = { status: 'ok', data: sheet.getDataRange().getValues() };
    }

    // 取得請假申請
    else if (action === 'getLeave') {
      const sheet = ss.getSheetByName('請假申請');
      result = { status: 'ok', data: sheet.getDataRange().getValues() };
    }

    // 取得加班申請
    else if (action === 'getOvertime') {
      const sheet = ss.getSheetByName('加班申請');
      result = { status: 'ok', data: sheet.getDataRange().getValues() };
    }

    // 取得補打卡申請
    else if (action === 'getSupp') {
      const sheet = ss.getSheetByName('補打卡申請');
      result = { status: 'ok', data: sheet.getDataRange().getValues() };
    }

    // 取得員工自己的審批狀態
    else if (action === 'getMyStatus') {
      const empId = params.empId || e.parameter.empId;
      result = handleGetMyStatus(ss, empId);
    }

    // ★ 自助診斷：直接在瀏覽器網址列打開 GAS_URL + ?action=diag 即可查看，
    //   不用看程式碼也能確認「工作表名稱有沒有打對」「欄位順序對不對」。
    else if (action === 'diag') {
      result = handleDiag(ss);
    }

  } catch (err) {
    result = { status: 'error', message: err.toString() };
  }

  if (callback) {
    return ContentService
      .createTextOutput(callback + '(' + JSON.stringify(result) + ')')
      .setMimeType(ContentService.MimeType.JAVASCRIPT);
  }
  return respond(result);
}

/**
 * 共用登入邏輯 (微調版)
 */
function handleLogin(ss, data) {
  const sheet = ss.getSheetByName('員工名單');
  const rows = sheet.getDataRange().getValues();
  for (let i = 1; i < rows.length; i++) {
    const [empId, name, password, role] = rows[i];
    if (String(empId) === String(data.empId) && String(password) === String(data.password)) {
      // 確保這裡有正確回傳 name, role, empId
      return { status: 'ok', name: name, role: role, empId: empId };
    }
  }
  return { status: 'error', message: '帳號或密碼錯誤' };
}

/**
 * 共用審批邏輯（doGet / doPost 都呼叫這支，確保兩個入口行為一致）
 * 比對策略：
 *   1) 優先用 clientId 精準比對
 *   2) 找不到 clientId 對應欄位或沒比對到時，fallback 用 empId + 狀態仍為「待審」模糊比對
 * 欄位位置一律查 SHEET_COLUMNS，不再用「動態抓最後一欄」去推算（那個算法之前算錯了一欄，
 * 導致核准/拒絕結果寫到 clientId 欄、狀態欄反而沒更新）。
 * 回傳 found:true/false，讓前端知道是否真的寫入了 Sheet。
 */
function handleApprove(ss, data) {
  const typeMap = {
    '請假': '請假申請',
    '加班': '加班申請',
    '補打卡': '補打卡申請'
  };
  const sheetName = typeMap[data.type];
  if (!sheetName) {
    return { status: 'error', message: '未知的申請類型：' + data.type };
  }

  const sheet = ss.getSheetByName(sheetName);
  if (!sheet) {
    return { status: 'error', message: '找不到工作表：' + sheetName };
  }

  const cols = SHEET_COLUMNS[sheetName];
  const statusIdx0 = cols.status - 1;     // 轉成 0-indexed，用來讀 getValues() 回來的陣列
  const clientIdIdx0 = cols.clientId - 1;

  const rows = sheet.getDataRange().getValues();
  let targetRow = -1;

  // 1) 用 clientId 精準比對
  if (data.clientId) {
    for (let i = 1; i < rows.length; i++) {
      const rowClientId = rows[i][clientIdIdx0];
      if (rowClientId && String(rowClientId) === String(data.clientId)) {
        targetRow = i + 1;
        break;
      }
    }
  }

  // 2) fallback：用 empId + 狀態仍為「待審」模糊比對（取最後一筆，避免比對到舊紀錄）
  if (targetRow === -1) {
    for (let i = rows.length - 1; i >= 1; i--) {
      const rowEmpId = rows[i][1];
      const rowStatus = rows[i][statusIdx0];
      if (String(rowEmpId) === String(data.empId) && rowStatus === '待審') {
        targetRow = i + 1;
        break;
      }
    }
  }

  if (targetRow === -1) {
    return { status: 'error', message: '找不到對應的申請列，狀態未更新', found: false };
  }

  // 寫入審核結果，以及核准/拒絕的主管姓名
 sheet.getRange(targetRow, cols.status).setValue(data.decision);
sheet.getRange(targetRow, cols.approver).setValue(data.approverName || data.approverId || '');
sheet.getRange(targetRow, cols.approveTime).setValue(data.approveTime ? new Date(data.approveTime) : new Date());

  return { status: 'ok', found: true, row: targetRow };
}

/**
 * 取得員工自己的審批狀態（含補打卡）
 */
function handleGetMyStatus(ss, empId) {
  const leaveSheet = ss.getSheetByName('請假申請');
  const otSheet = ss.getSheetByName('加班申請');
  const suppSheet = ss.getSheetByName('補打卡申請');

  const leaveRows = leaveSheet.getDataRange().getValues().slice(1);
  const otRows = otSheet.getDataRange().getValues().slice(1);
  const suppRows = suppSheet ? suppSheet.getDataRange().getValues().slice(1) : [];

  const updates = [];

  leaveRows.filter(r => String(r[1]) === String(empId)).forEach(r => {
    // J(9)=狀態, K(10)=clientId
    const clientId = r[10] || (r[1] + '_leave_' + r[4]);
    updates.push({ type: '請假', date: r[4], status: r[9], clientId: String(clientId) });
  });

  otRows.filter(r => String(r[1]) === String(empId)).forEach(r => {
    const clientId = r[9] || (r[1] + '_ot_' + r[3]);
    updates.push({ type: '加班', date: r[3], status: r[8], clientId: String(clientId) });
  });

  suppRows.filter(r => String(r[1]) === String(empId)).forEach(r => {
    const clientId = r[8] || (r[1] + '_supp_' + r[4]);
    updates.push({ type: '補打卡', date: r[4], status: r[7], clientId: String(clientId) });
  });

  return { status: 'ok', updates };
}

/**
 * 請假／加班／補打卡 有新申請時，寄信通知主管查閱審核。
 * 用 MailApp 寄信失敗不應該影響申請本身已經寫入 Sheet 這件事，
 * 所以這裡用 try/catch 包起來，寄信失敗只記錄錯誤、不往外拋例外。
 */
function notifyAdmins(type, data) {
  try {
    const who = `${data.name || ''}（員編 ${data.empId || ''}）`;
    const subject = `[TJCPM] 新${type}申請待審核 - ${who}`;

    const lines = [];
    lines.push(`員工：${who}`);
    lines.push(`申請類型：${type}`);

    if (type === '請假') {
      lines.push(`假別：${data.subType || ''}`);
      lines.push(`開始：${data.date || ''} ${data.startTime || ''}`);
      lines.push(`結束：${data.endDate || ''} ${data.endTime || ''}`);
      if (data.reason) lines.push(`原因：${data.reason}`);
    } else if (type === '加班') {
      lines.push(`日期：${data.date || ''}`);
      lines.push(`時間：${data.startTime || ''} ～ ${data.endTime || ''}`);
      if (data.hours) lines.push(`時數：${data.hours}`);
      if (data.reason) lines.push(`原因：${data.reason}`);
    } else if (type === '補打卡') {
      lines.push(`補打卡類型：${data.subType || ''}`);
      lines.push(`日期：${data.date || ''}`);
      lines.push(`時間：${data.time || ''}`);
      if (data.reason) lines.push(`原因：${data.reason}`);
    }

    // ★ 明確列出目前的審核狀態，剛送出的申請一律是「待審」，讓主管一看信就知道要處理
    lines.push(`審核狀態：待審`);

    lines.push('');
    lines.push('登入簽核：https://tjcpm109.github.io/tjcpm/')
    lines.push('請登入系統查閱：');
    lines.push(SpreadsheetApp.getActiveSpreadsheet().getUrl());

    MailApp.sendEmail({
      to: ADMIN_NOTIFY_EMAILS.join(','),
      subject: subject,
      body: lines.join('\n')
    });
  } catch (err) {
    console.error('notifyAdmins 寄信失敗: ' + err.toString());
  }
}

/**
 * 安全寫入工作表：找不到該名稱的工作表，或寫入時發生其他錯誤，
 * 都只記錄到 Apps Script 的「執行紀錄 / Executions」log，不會讓整個請求中斷。
 * 這樣即使 Sheet 寫入失敗，後面的 notifyAdmins() 還是會被執行到，
 * 方便用「有沒有收到信」來判斷問題是出在 Sheet 設定、還是請求根本沒送到伺服器。
 *
 * ⚠️ 注意：前端送出申請時用的是 fetch(..., {mode:'no-cors'})，這種模式下
 * 瀏覽器端完全看不到伺服器回應內容（即使是錯誤訊息），所以這裡的錯誤只會留在
 * Apps Script 後台，不會顯示在網頁畫面上。要查錯誤，請到 Apps Script 編輯器
 * 左側選單「執行作業 / Executions」，點進失敗的那筆查看詳細錯誤訊息。
 */
function appendToSheetSafely(ss, sheetName, rowValues) {
  try {
    const sheet = ss.getSheetByName(sheetName);
    if (!sheet) {
      console.error(`appendToSheetSafely 失敗：找不到工作表「${sheetName}」，請確認 Google Sheet 裡有一個分頁名稱完全是「${sheetName}」（注意全形/半形、有無多餘空白）。`);
      return false;
    }
    sheet.appendRow(rowValues);
    return true;
  } catch (err) {
    console.error(`appendToSheetSafely 寫入「${sheetName}」時發生錯誤：${err.toString()}`);
    return false;
  }
}

/**
 * 自助診斷：列出這個 Google Sheet 裡所有分頁名稱、欄數、以及第一列（標題列）內容。
 * 用法：直接在瀏覽器網址列打開
 *   你的GAS_URL?action=diag
 * 例如部署網址是 https://script.google.com/macros/s/xxx/exec，
 * 就打開 https://script.google.com/macros/s/xxx/exec?action=diag
 * 用來確認：
 *   1) 分頁名稱是否跟程式碼裡用的完全一致（員工名單／打卡紀錄／請假申請／加班申請／補打卡申請）
 *   2) 每個分頁目前的欄數、標題內容，是否跟 SHEET_COLUMNS 設定的欄位編號對得上
 */
function handleDiag(ss) {
  const sheets = ss.getSheets();
  const info = sheets.map(s => {
    const name = s.getName();
    const lastCol = s.getLastColumn();
    const lastRow = s.getLastRow();
    const headerRow = lastCol > 0 ? s.getRange(1, 1, 1, lastCol).getValues()[0] : [];
    return { name, lastCol, lastRow, headerRow };
  });
  const expectedSheets = ['員工名單', '打卡紀錄', '請假申請', '加班申請', '補打卡申請'];
  const actualNames = info.map(s => s.name);
  const missing = expectedSheets.filter(n => !actualNames.includes(n));
  return {
    status: 'ok',
    spreadsheetUrl: ss.getUrl(),
    sheets: info,
    missingExpectedSheets: missing,
    expectedColumnLayout: SHEET_COLUMNS
  };
}

function respond(obj) {
  return ContentService
    .createTextOutput(JSON.stringify(obj))
    .setMimeType(ContentService.MimeType.JSON);
}
